148 void* list_pop(list_t* list)
149 {
150  struct list_node* old_anchor;
151  void* value;
152 
153  if (!list) {
154  return NULL;
155  }
156 
157  if (list->anchor) {
158  value = list->anchor->value;
159 
160  old_anchor = list->anchor;
161  list->anchor = list->anchor->next;
162  delete old_anchor;
163 
164  list->size--;
165  } else {
166  value = NULL;

174  }

197 static struct list_node* list_get_node_at(list_t* list, int index)
198 {
199  if (!list || !list->anchor || (index < 0)) {
200  return NULL;
201  }
202  //Chaching deaktivieren
203  //last_list = NULL;
204  struct list_node* current = list->anchor;
205  int n = index;
206 
207  if (last_list && (last_list == list) && (last_index <= index) && (last_node != NULL)) {
208  n -= last_index;
209  current = last_node;
210  }
211 
212  while (n--) {
213  current = current->next;
214  if (!current) {
215  return NULL;
216  }
217  }
218 
219  last_list = list;
220  last_index = index;
221  last_node = current;
222 
223  return current;
224 }
