87 int
88 err_to_errno(err_t err)
89 {
90  if ((err > 0) || (-err >= (err_t)LWIP_ARRAYSIZE(err_to_errno_table))) {
91  return EIO;
92  }
93  return err_to_errno_table[-err];
94 }
