115 bool Display::setScreenMode(size_t modeId)
116 {
117  Display::ScreenMode *pSm = 0;
118 
119  List<Display::ScreenMode *> modes;
120  if (!getScreenModes(modes))
121  return false;
122  for (List<Display::ScreenMode *>::Iterator it = modes.begin();
123  it != modes.end(); it++)
124  {
125  if ((*it)->id == modeId)
126  {
127  pSm = *it;
128  break;
129  }
130  }
131  if (pSm == 0)
132  {
133  ERROR("Screenmode not found: " << modeId);
134  return false;
135  }
136 
137  return setScreenMode(*pSm);
138 }
